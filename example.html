jQuery POP'n SocialButton サンプル

サンプル1: githubのシェアボタン

    $('.social-button.github').popnSocialButton([ 'twitter', 'facebook', 'hatebu' ], {
      url: 'http://github.com/'
    });

サンプル2: googleのシェアボタン

    $('.social-button.google').popnSocialButton([ 'facebook', 'twitter' ], {
      url: 'http://www.google.com/',
      text: 'ググレカス', 
      imgDir: './img',
      buttonSpace: 64,
      countPosition: {
        top: 8,
        left: 50
      },
      countForeColor: '#222222',
      countBackColor: '#eeeeee',
      countBorderColor: '#888888',
      countSize: 16
    });
